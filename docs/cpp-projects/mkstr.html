mkstr, c++

mkstr is a c++ string library written in pure c++ modules

Github: https://github.com/fitban/mkstr

import std;
import mkstr.string;

int main()
{
	mkstr::string str{"Cheers, c++!"};
	std::cout
		<<  str << '\n'
		<< str.cpp_str() << '\n'
		<< str.raw_str() << '\n'
		<< str.str() << '\n'
		<< str.data() << '\n'
	;
}
